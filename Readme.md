# rexfeng ruby gem (1.0.2)

Contact info Ruby Gem for Rex Feng

# Installation
    gem install rexfeng

# Example Usage
    require 'rexfeng'

    Rexfeng.name    #=> 'Rex Feng'
    Rexfeng.github  #=> 'https://github.com/xta'
